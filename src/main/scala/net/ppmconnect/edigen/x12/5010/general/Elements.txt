package net.ppmconnect.edigen.x12.general
package elements

/**
 * The "X12.3 Data Element Dictionary" for 5010 encoded in Scala
 */

sealed class ElementUsage[+T <: definitions.ElementDefinition](
  val elementDefinition: T,
  val repeat: designations.repeat.Repeat,
  val requirement: designations.requirement.Requirement,
  val parent: ElementGroup,
  val index: Int) {
}

/**
 * Used by containers of elements: segments and composite elements. Note that SegmentGroup
 * is defined in TransactionSet.scala (instead of Segments.scala), because segment groups
 * only exist within the context of a transaction set. Element groups are different, because
 * they are only defined within the context of a segment or composite element -- both of
 * which exist independently of a transaction set.
 */
trait ElementGroup {
  import scala.collection.mutable.ListBuffer
  import definitions.ElementDefinition
  import designations.requirement.Requirement
  import designations.syntax.Syntax
  import designations.repeat.Repeat

  type E = ElementUsage[ElementDefinition]

  /**
   * @todo: This design forces us to lose the singleton ElementDefinition type, like DE1000
   * instead only permitting /some/ ElementDefinition. It's not clear at this point if that
   * type information needs to be recovered when iterating an ElementGroup.
   */
  private val elementUsages: ListBuffer[E] = new ListBuffer()

  /**
   * If `this` is a CompositeElementDefinition, the composite element name. Otherwise
   * `this` is a SegmentDefinition and elementGroupId is the segment name.
   */
  def elementGroupId: String

  /**
   * Returns a list of each element and its usage attributes (repeat, requirement, etc)
   */
  def elements = elementUsages.toList

  /**
   * Declare the usage of an element and bind it to the ElementGroup `this`.
   *
   * @todo: If the requirement designation is X, ensure that a Syntax was defined that references this element.
   */
  def Element[T <: ElementDefinition](elementDefinition: T, repeat: Repeat, requirement: Requirement): ElementUsage[T] = {
    val usage = new ElementUsage(elementDefinition, repeat, requirement, this, elementUsages.length + 1)
    elementUsages += (usage: E)
    return usage
  }

  /**
   * Declares the "syntax", or dependencies, between elements in the same ElementGroup. Some
   * of these can be analyzed statically, while most of them are likely only determinable at
   * runtime, depending on the presence or absence of an element.
   *
   * @todo: Prevent declaring dependencies on elements from disparate ElementGroups.
   */
  def Syntax(syntax: Syntax*) { }
}

/**
 * The abbreviated type names are used as class names -- not because it make the code
 * look unreadable, but because these match their use in the transaction set guides,
 * making it actually easier to read the code if you have the documentation available.
 *
 * Also, it would be annoying to import this package and have it mess with the Scala
 * types of the same names (eg: String)
 */
package types {
  import definitions.SimpleElementDefinition

  /**
   * Unspecified type, only "used" in the ISA control segment
   */
  sealed abstract class NA(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)

  /**
   * Numeric (pattern: <-?[0-9]+>)
   *   The value of a numeric element includes an implied decimal point. It is used when the
   *   position of the decimal point within the data is permanently fixed and is not transmitted
   *   with the data. The data element dictionary defines the implied number of decimal positions.
   *   For negative values, the leading minus sign (-) is used. Absence of a sign indicates a positive
   *   value.  The plus sign (+) should not be transmitted. Leading zeros should be supressed
   *   unless necessary to satisfy a minimum length requirement. The length of a numeric type element
   *   does not include the optional sign.
   */
  sealed abstract class Nn(description: String, minLength: Int, maxLength: Int, decimalPlaces: Int)
    extends SimpleElementDefinition(description, minLength, maxLength) { require(decimalPlaces >= 0) }

  sealed abstract class N0(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 0)

  sealed abstract class N1(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 1)

  sealed abstract class N2(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 2)

  sealed abstract class N3(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 3)

  sealed abstract class N4(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 4)

  sealed abstract class N5(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 5)

  sealed abstract class N6(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 6)

  sealed abstract class N7(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 7)

  sealed abstract class N8(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 8)

  sealed abstract class N9(description: String, minLength: Int, maxLength: Int)
    extends Nn(description, minLength, maxLength, decimalPlaces = 9)

  /**
   * Decimal (pattern: <-?(([0-9]+(\.[0-9]*)?)|\.[0-9]*)(E-?[0-9]+)?>)
   *   The decimal element contains an explicit decimal point and is used for numeric values that
   *   have a varying number of decimal positions. The decimal point always appears in the character
   *   stream if the decimal point is at any place other than the right end. If the value is an
   *   integer (decimal point at the right end), the decimal point should be omitted. For negative
   *   values, the leading minus sign (-) is used. Absence of a sign indicates a positive value. The
   *   plus sign (+) shall not be transmitted. Leading zeros following the decimal point should be
   *   supressed unless necessary to satisfy a minimum length requirement. Trailing zeros following
   *   the decimal point should be supressed unless necessary to indicate precision. A base 10 exponent
   *   may be appended at the end of a decimal number. The use of triad separators (eg: 1,00,000) is
   *   expressly prohibited. The length of a decimal type element does not include the optional minus
   *   signs, decimal point, or exponent indicator E.
   */
  sealed abstract class R(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)

  /**
   * Identifier (pattern: <[0-9A-Z]+ *>)
   *   The identifier element always contains a unique value from a single, predefined list of
   *   values. Trailing spaces should be supressed.
   */
  sealed abstract class ID(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)

  /**
   * String (pattern: <[[:nonspace] ]*[[:nonspace]][[:nonspace] ]*>)
   *   The string element is a sequence of any characters from the basic or extended character
   *   sets, and it contains at least one non-space character. The significant characters shall be
   *   left-justified. Leading spaces, when they occur, are presumed to be significant characters.
   *   In the actual data stream, trailing characters should be supressed.
   */
  sealed abstract class AN(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)

  /**
   * Date (pattern: <([0-9]{2}|[0-9]{4})("01" | ... "12")("01" | ... | "31")>
   *   The date element is used to express the standard date in either YYMMDD or CCYYMMDD format,
   *   in which the CC is the first two digits of the calendar year, YY is the last two digits
   *   of the calendar year, MM is the month, and DD is the day in the month.
   */
  sealed abstract class DT(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)

  /**
   * Time (pattern: <("00" | ... | "23")("00" | ... | "59")(("00" | ... | "59")[0-9]*)?>
   *   The time element is used to express time in HHMMSSd..d format, in which HH is the hour for a
   *   24-hour clock, MM is the minute, SS is the seconds, and d...d is the fractional seconds.
   *   Seconds and fractional seconds are optional. Trailing zeros should be suppressed unless
   *   necessary to satisfy a minimum length requirement or unless necessary to indicate precision.
   */
  sealed abstract class TM(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)

  /**
   * Binary
   *   The binary element is any sequence of octects ranging from 00000000 to 11111111 binary (0 to
   *   255 in decimal notation). The element type has no defined maximum length. Actual length is
   *   specified by the immediately preceeding element. The binary element type may only exist in the
   *   Binary and Security header segments.
   */
  sealed abstract class B(description: String, minLength: Int, maxLength: Int)
    extends SimpleElementDefinition(description, minLength, maxLength)
}

package definitions {
  import types._

  sealed abstract class ElementDefinition(val description: String) {

    /**
     * Calculates the element ID, like DE19, based on the singleton name. It
     * doesn't work if `this` is not a singleton instance (object).
     */
    def elementId: String = {
      var name = this.getClass.getName
      if (name endsWith "$")
        name = name.substring(0, name.length - 1)

      val m = name.lastIndexOf('.': Int)
      if (m >= 0)
        name = name.substring(m + 1)

      val n = name.indexOf('$': Int)
      if (n >= 0)
        name = name.substring(n + 1)

      name
    }

    override def toString = elementId + ": " + description
  }

  /**
   * Simple elements are the smallest unit of data (think native types)
   */
  sealed abstract class SimpleElementDefinition(description: String, minLength: Int, maxLength: Int) extends ElementDefinition(description) {
    require(minLength >  0)
    require(maxLength >= maxLength)
  }

  /**
   * Composite elements consist of two or more elements preceeded by a data element separator. In
   * use, in the actual data stream, a composite element may appear as only one component element.
   * Each component element within the composite element, except the last, is followed by a component
   * element separator. The final component element is followed by the next element separator or the
   * segment terminator. Trailing component element separators shall be supressed. Composite elements
   * are defined in a directory. The directory defines each composite element by its name, purpose,
   * reference identifier, and included component elements in a specified sequence.
   *
   * @todo: Check for at least two simple element definitions
   */
  sealed abstract class CompositeElementDefinition(description: String) extends ElementDefinition(description) with ElementGroup {
    def elementGroupId = elementId
  }
